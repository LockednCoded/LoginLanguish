cmake_minimum_required(VERSION 3.3.2)
project(LanguishTest)
set(CMAKE_CXX_STANDARD 11)

add_library(webview libs/webview/webview.cc libs/webview/webview.h)
add_executable(app basic.cpp)

# Include libs directory
target_include_directories(app PRIVATE libs)

# Link against the necessary frameworks or libraries
target_link_libraries(app
    "-framework WebKit"
    ${CMAKE_SOURCE_DIR}/libs/webview/webview.cc
)

# Set the output directory for the built executable
set_target_properties(app PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}
)
